using System;

public class Program
{

        if ( disponi == false)

        }
      else
      if (categoria == 'A'|| categoria == 'a' && precioProucto < 100)

         Console.WriteLine(" El precio del producto para esta categoría debe ser mayor a 100");
      }

        {
            descue = precioProucto - (precioProucto * 0.5);
            Console.WriteLine(" El nombre del producto es: {0}", nombreProducto);

            Console.WriteLine(" La disponibilidad es: {0}", disponi);
            Console.WriteLine(" La categoria del producto es: {0} y el precio del producto más el descuento es: {1}", nombreProducto, descue);
        }
        else
          if (categoria == 'B' || categoria == 'b' && precioProucto > 50)
         {
         Console.WriteLine(" El precio del producto para esta categoría debe ser menor a 50.");
          }
          else if (categoria == 'C' || categoria == 'c')

            Console.WriteLine(" El nombre del producto es: {0}", nombreProducto);
